"use client";

import { useState, useEffect, useRef } from "react";
import { useRouter } from "next/navigation";
import Image from "next/image";
import { auth } from "@/lib/firebase";
import { useCredits } from "@/contexts/CreditContext";
import { useImageUpload } from "@/hooks/useImageUpload";
import { generateImage } from "@/lib/api";
import { ROUTES } from "@/utils/routes";
import { STYLE_CATEGORIES, PLACEHOLDER_IMAGE, APP_LOGO_WHITE, PREMIUM_ICON } from "@/constants";
import toast from "react-hot-toast";

export default function GeneratorScreen() {
  const router = useRouter();
  const { credits, isPremium, isLoading: creditsLoading, decrementCredits } = useCredits();
  const { selectedImage, imagePreview, handleImageSelect, clearImage, isValidImage } = useImageUpload();
  const fileInputRef = useRef<HTMLInputElement>(null);

  const [desiredStyle, setDesiredStyle] = useState<string>("");
  const [famousStyle, setFamousStyle] = useState<string>("");
  const [luxuryStyle, setLuxuryStyle] = useState<string>("");
  const [powerStyle, setPowerStyle] = useState<string>("");
  const [strengthStyle, setStrengthStyle] = useState<string>("");
  const [fantasyStyle, setFantasyStyle] = useState<string>("");
  const [freedomStyle, setFreedomStyle] = useState<string>("");
  const [chaosStyle, setChaosStyle] = useState<string>("");

  const [isGenerating, setIsGenerating] = useState(false);
  const [generatedImageUrl, setGeneratedImageUrl] = useState<string | null>(null);
  const [showMobileMenu, setShowMobileMenu] = useState(false);

  useEffect(() => {
    const user = auth.currentUser;
    if (!user) {
      router.replace(ROUTES.LOGIN);
    }
  }, [router]);

  const getSelectedStyle = () => {
    const styles = [desiredStyle, famousStyle, luxuryStyle, powerStyle, strengthStyle, fantasyStyle, freedomStyle, chaosStyle];
    const selected = styles.find(s => s !== "");
    if (!selected) return null;
    for (const category of Object.values(STYLE_CATEGORIES)) {
      const style = category.options.find((s) => s.key === selected);
      if (style) return style;
    }
    return null;
  };

  const handleGenerate = async () => {
    const user = auth.currentUser;
    if (!user) {
      toast.error("Please log in to generate images");
      router.replace(ROUTES.LOGIN);
      return;
    }
    if (!selectedImage) {
      toast.error("Please upload an image");
      return;
    }
    const styleInfo = getSelectedStyle();
    if (!styleInfo) {
      toast.error("Please select a style");
      return;
    }
    if (styleInfo.isPremium && !isPremium) {
      toast.error("This style requires Premium. Upgrade to unlock!");
      router.push(ROUTES.PREMIUM);
      return;
    }
    if (!isPremium && credits < 1) {
      toast.error("No credits remaining. Purchase more or upgrade to Premium!");
      router.push(ROUTES.PREMIUM);
      return;
    }
    setIsGenerating(true);
    try {
      const formData = new FormData();
      formData.append("image1", selectedImage);
      formData.append("prompt", styleInfo.prompt);
      formData.append("userId", user.uid);
      const imageBlob = await generateImage(formData);
      if (!isPremium) {
        await decrementCredits(1);
      }
      const imageUrl = URL.createObjectURL(imageBlob);
      setGeneratedImageUrl(imageUrl);
      toast.success("Image generated successfully!");
    } catch (error) {
      console.error("Generation error:", error);
      toast.error("Failed to generate image. Please try again.");
    } finally {
      setIsGenerating(false);
    }
  };

  const downloadImage = () => {
    if (!generatedImageUrl) return;
    const link = document.createElement("a");
    link.href = generatedImageUrl;
    link.download = `dreamai-\${Date.now()}.png`;
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
    toast.success("Image downloaded!");
  };

  const shareImage = async () => {
    if (!generatedImageUrl) return;
    try {
      const response = await fetch(generatedImageUrl);
      const blob = await response.blob();
      const file = new File([blob], "dreamai-image.png", { type: "image/png" });
      if (navigator.share && navigator.canShare({ files: [file] })) {
        await navigator.share({
          files: [file],
          title: "DreamAI Generated Image",
        });
        toast.success("Image shared!");
      } else {
        toast.error("Sharing not supported on this device");
      }
    } catch (error) {
      console.error("Share error:", error);
      toast.error("Failed to share image");
    }
  };

  if (creditsLoading) {
    return (
      <div className="flex items-center justify-center min-h-screen bg-white">
        <div className="animate-spin rounded-full h-12 w-12 border-t-2 border-b-2 border-brand"></div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-white text-black flex flex-col">
      <header className="bg-white border-b border-gray-200 px-6 py-4">
        <div className="max-w-4xl mx-auto flex items-center justify-between">
          <div className="flex items-center gap-3">
            <button onClick={() => setShowMobileMenu(!showMobileMenu)} className="p-2 hover:bg-gray-100 rounded-lg transition-colors">
              <svg xmlns="http://www.w3.org/2000/svg" className="h-6 w-6" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M4 6h16M4 12h16M4 18h16" />
              </svg>
            </button>
            <h1 className="text-xl font-bold">DreamAI</h1>
          </div>
          <div className="flex items-center gap-3">
            {!isPremium && (<div className="text-sm font-semibold">{credits} Credits</div>)}
            <button onClick={() => router.push(ROUTES.PREMIUM)} className="bg-brand hover:bg-brand-dark text-white px-4 py-2 rounded-full flex items-center gap-2 transition-colors">
              <Image src={PREMIUM_ICON} alt="Premium" width={16} height={16} />
              <span className="text-sm font-semibold">PREMIUM</span>
            </button>
          </div>
        </div>
        {showMobileMenu && (
          <div className="mt-4 border-t border-gray-200 pt-4">
            <nav className="flex flex-col gap-2">
              <button onClick={() => { router.push(ROUTES.PREMIUM); setShowMobileMenu(false); }} className="text-left px-4 py-2 hover:bg-gray-100 rounded-lg transition-colors">Premium</button>
              <button onClick={() => { router.push(ROUTES.SETTINGS); setShowMobileMenu(false); }} className="text-left px-4 py-2 hover:bg-gray-100 rounded-lg transition-colors">Settings</button>
            </nav>
          </div>
        )}
      </header>
      <main className="flex-1 overflow-y-auto px-6 py-6">
        <div className="max-w-3xl mx-auto">
          <div className="mb-6">
            <div className="flex items-center gap-2 mb-3">
              <svg xmlns="http://www.w3.org/2000/svg" className="h-5 w-5" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M4 16l4.586-4.586a2 2 0 012.828 0L16 16m-2-2l1.586-1.586a2 2 0 012.828 0L20 14m-6-6h.01M6 20h12a2 2 0 002-2V6a2 2 0 00-2-2H6a2 2 0 00-2 2v12a2 2 0 002 2z" />
              </svg>
              <h2 className="text-base font-semibold">Face Photo</h2>
              <button className="ml-1 w-5 h-5 bg-gray-200 rounded-full flex items-center justify-center text-xs text-gray-600">?</button>
            </div>
            <div className="relative">
              <input ref={fileInputRef} type="file" accept="image/*" onChange={handleImageSelect} className="hidden" />
              <div onClick={() => { if (!imagePreview && fileInputRef.current) { fileInputRef.current.click(); } }} className="w-full h-32 bg-gray-50 border-2 border-dashed border-gray-300 rounded-xl cursor-pointer hover:border-brand transition-colors overflow-hidden flex items-center justify-center">
                {imagePreview ? (
                  <div className="relative w-full h-full">
                    <Image src={imagePreview} alt="Preview" fill className="object-cover" />
                    <button onClick={(e) => { e.preventDefault(); e.stopPropagation(); clearImage(); }} className="absolute top-2 right-2 bg-black/60 p-1.5 rounded-full hover:bg-black/80">
                      <svg xmlns="http://www.w3.org/2000/svg" className="h-4 w-4 text-white" viewBox="0 0 20 20" fill="currentColor">
                        <path fillRule="evenodd" d="M4.293 4.293a1 1 0 011.414 0L10 8.586l4.293-4.293a1 1 0 111.414 1.414L11.414 10l4.293 4.293a1 1 0 01-1.414 1.414L10 11.414l-4.293 4.293a1 1 0 01-1.414-1.414L8.586 10 4.293 5.707a1 1 0 010-1.414z" clipRule="evenodd" />
                      </svg>
                    </button>
                  </div>
                ) : (
                  <div className="flex flex-col items-center justify-center text-gray-400">
                    <Image src={PLACEHOLDER_IMAGE} alt="Upload" width={40} height={40} className="mb-2 opacity-50" />
                    <p className="text-sm">Upload photo</p>
                  </div>
                )}
              </div>
            </div>
          </div>
